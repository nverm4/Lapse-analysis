### Création de nouvelles variables


## Préparation du data
data_acp <- data
levels(data_acp$polholder_diffdriver) <- c(rep("autre", 4),
                                           "same", rep("autre", 2))     # pour avoir 2 groupes
nom_retire <- c("lapse", "policy_nbcontract", "prem_freqperyear", "vehicl_agepurchase",
                "vehicl_garage", "vehicl_region", "policy_caruse")
data_acp <- data_acp[, !(colnames(data) %in% nom_retire)]


# création des indicatrices
data_acp <- dummy_cols(data_acp, remove_most_frequent_dummy = TRUE)
nom_retire_ind <- c("polholder_BMCevol", "polholder_gender", "polholder_job",
                    "vehicl_powerkw", "polholder_diffdriver")
data_acp <- data_acp[, !(colnames(data_acp) %in% nom_retire_ind)]


# création des variables des différences
data_acp$Diff_last_vs_final <- (data_acp$prem_last - data_acp$prem_final)/
    data_acp$prem_last
# stable = variation +/- 1 %

data_acp$Diff_final_vs_market <- (data_acp$prem_final - data_acp$prem_market)/
    data_acp$prem_final
# stable = variation +/- 1 %

data_acp$rabais <- as.numeric((data_acp$prem_final - data_acp$prem_pure)/
    data_acp$prem_final < 0)
# si < 0, sûrement un rabais
